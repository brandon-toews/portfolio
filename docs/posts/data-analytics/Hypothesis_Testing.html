F-Tests

First, run an F-test on the untransformed positively skew data

# Extract 'avg_d_mbps' for the fixed and mobile networks of the yeo-j transformed
fixed_avg_d_mbps = dfs['Fixed']['avg_d_mbps']
mobile_avg_d_mbps = dfs['Mobile']['avg_d_mbps']

# Calculate the sample variances for fixed and mobile
fixed_variance = np.var(fixed_avg_d_mbps, ddof=1)
mobile_variance = np.var(mobile_avg_d_mbps, ddof=1)

# Perform the F-test
f_statistic = fixed_variance / mobile_variance
df1 = len(fixed_avg_d_mbps) - 1
df2 = len(mobile_avg_d_mbps) - 1

# Calculate two-tailed p-value by calculating the CDF of the
# F-distribution at the f_statistic
p_value = 2 * (1 - stats.f.cdf(np.abs(f_statistic), df1, df2))

# Display results
print('F statistic:', f_statistic)
print('p-value:', np.round(p_value, 10))

# Print which network has larger variance
if f_statistic > 1:
    print("The fixed network has a larger variance.")
elif f_statistic < 1:
    print("The mobile network has a larger variance.")
else:
    print("The variances are equal.")

Man-Whitney U Test

Run the untransformed data of this non-parametric test

# Extract 'avg_d_mbps' for the fixed and mobile networks
fixed_avg_d_mbps = dfs['Fixed']['avg_d_mbps']
mobile_avg_d_mbps = dfs['Mobile']['avg_d_mbps']

# Perform the Mann-Whitney U test
u_statistic, p_value = stats.mannwhitneyu(fixed_avg_d_mbps, mobile_avg_d_mbps, alternative='two-sided')

# Check the medians of the two groups
# whichever median is higher tends to have larger speeds
if np.median(fixed_avg_d_mbps) > np.median(mobile_avg_d_mbps):
    print("The fixed network tends to have larger download speeds.")
elif np.median(fixed_avg_d_mbps) < np.median(mobile_avg_d_mbps):
    print("The mobile network tends to have larger download speeds.")
else:
    print("The download speeds are similar for both networks.")

# Display results
print('U statistic:', u_statistic)
print('p-value:', p_value)
